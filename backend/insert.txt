-- 先刪除所有資料（注意刪除順序很重要，因為有外鍵約束）
DELETE FROM transmitter;
DELETE FROM receiver;
DELETE FROM device;

-- 重設序列（如果有使用自增ID）
ALTER SEQUENCE device_id_seq RESTART WITH 1;

-- 插入設備數據 - 使用枚舉值而非字符串
INSERT INTO device (name, device_type, x, y, z, active) VALUES
('tx_main', 'TRANSMITTER', 0.0, 60.0, 2.0, true),
('tx_i', 'TRANSMITTER', -100.0, 100.0, 2.0, true),
('rx', 'RECEIVER', 0.0, 0.0, 1.5, true);

-- 插入發射器數據
INSERT INTO transmitter (id, transmitter_type)
SELECT id, 'SIGNAL' FROM device WHERE name = 'tx_main';

INSERT INTO transmitter (id, transmitter_type)

-- 插入接收器數據
INSERT INTO receiver (id)
